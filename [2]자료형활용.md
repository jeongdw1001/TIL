#아래 코드와 동일(docs에서 가져옴, 참고만)   

def to_bytes(n, length=1, byteorder='big', signed=False):
    if byteorder == 'little':
        order = range(length)
    elif byteorder == 'big':
        order = reversed(range(length))
    else:
        raise ValueError("byteorder must be either 'little' or 'big'")
    return bytes((n >> i*8) & 0xff for i in order)
